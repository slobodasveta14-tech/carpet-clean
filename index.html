const canvas = document.getElementById("canvas"); const ctx = canvas.getContext("2d"); const stage = document.getElementById("stage"); const percentText = document.getElementById("percent"); const progress = document.getElementById("progress"); const finish = document.getElementById("finish"); const extractor = document.getElementById("extractor"); const dirty = new Image(); dirty.src = "До.jpg"; let drawing = false; let finished = false; let cleaned = new Set(); const cell = 22; function resizeCanvas(){ const r = stage.getBoundingClientRect(); canvas.width = r.width; canvas.height = r.height; ctx.globalCompositeOperation = "source-over"; ctx.clearRect(0,0,canvas.width,canvas.height); ctx.drawImage(dirty,0,0,canvas.width,canvas.height); } dirty.onload = resizeCanvas; window.addEventListener("resize", resizeCanvas); function getPos(e){ const r = canvas.getBoundingClientRect(); const p = e.touches ? e.touches[0] : e; return { x:p.clientX-r.left, y:p.clientY-r.top }; } function clean(x,y){ const brush = canvas.width * 0.14; const g = ctx.createRadialGradient(x,y,brush*0.1,x,y,brush); g.addColorStop(0,"rgba(0,0,0,1)"); g.addColorStop(0.7,"rgba(0,0,0,.9)"); g.addColorStop(1,"rgba(0,0,0,0)"); ctx.globalCompositeOperation = "destination-out"; ctx.fillStyle = g; ctx.beginPath(); ctx.arc(x,y,brush,0,Math.PI*2); ctx.fill(); ctx.globalCompositeOperation = "source-over"; markCleaned(x,y,brush); } function markCleaned(x,y,brush){ const cols = Math.ceil(canvas.width / cell); const rows = Math.ceil(canvas.height / cell); const minX = Math.max(0, Math.floor((x-brush)/cell)); const maxX = Math.min(cols-1, Math.floor((x+brush)/cell)); const minY = Math.max(0, Math.floor((y-brush)/cell)); const maxY = Math.min(rows-1, Math.floor((y+brush)/cell)); for(let gy=minY; gy<=maxY; gy++){ for(let gx=minX; gx<=maxX; gx++){ const cx = gx*cell + cell/2; const cy = gy*cell + cell/2; if(Math.hypot(cx-x, cy-y) < brush){ cleaned.add(gx + "-" + gy); } } } const total = cols * rows; let percent = Math.floor((cleaned.size / total) * 100); if(percent >= 85){ finishCleaning(); return; } percentText.textContent = percent; progress.style.width = percent + "%"; } function finishCleaning(){ if(finished) return; finished = true; ctx.globalCompositeOperation = "destination-out"; ctx.clearRect(0,0,canvas.width,canvas.height); ctx.globalCompositeOperation = "source-over"; percentText.textContent = "100"; progress.style.width = "100%"; finish.style.display = "block"; if(extractor) extractor.style.display = "none"; finish.scrollIntoView({behavior:"smooth", block:"nearest"}); } function move(e){ if(!drawing || finished) return; e.preventDefault(); const p = getPos(e); clean(p.x,p.y); if(extractor){ extractor.style.display = "flex"; extractor.style.left = p.x + "px"; extractor.style.top = p.y + "px"; } } canvas.addEventListener("mousedown", e => {drawing=true; move(e)}); canvas.addEventListener("mousemove", move); canvas.addEventListener("mouseup", () => drawing=false); canvas.addEventListener("mouseleave", () => drawing=false); canvas.addEventListener("touchstart", e => {drawing=true; move(e)}, {passive:false}); canvas.addEventListener("touchmove", move, {passive:false}); canvas.addEventListener("touchend", () => drawing=false);
